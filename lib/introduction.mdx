{/* 自定义 Canvas 画布( 放大/缩小/锚点标题等功能 )包裹 Story */}

<Canvas of={ExampleStories.DefaultExample} />

{/* 自定义 锚点副段落标题 */}

<Subheading>Another Example</Subheading>
